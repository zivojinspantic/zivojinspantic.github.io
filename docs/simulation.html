simm<-sim_sarima(n=64, model=list(sma=fit4$coef[1],ar= fit4$coef[2],

  geom_point(color='darkblue')+
  geom_line(color='darkblue')+
  theme_gray()+
  xlab('Time') +
  ylab('Value')+
  ggtitle("Simulation")
fig<-ggplotly(p)
